class Solution {
    public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
        if(l1 == null) return l2;
        if(l2 == null) return l1;

        ListNode dummy = new ListNode(-1);
        ListNode p = dummy;

        ListNode h1 = l1; //ä¸¤ä¸ªé“¾è¡¨çš„æŒ‡é’ˆ

                p.next = h2;
